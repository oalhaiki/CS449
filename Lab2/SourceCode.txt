package osama.alhaiki.umpirebuddy; import android.content.Context; import android.content.Intent; import android.support.v7.app.AppCompatActivity; import android.os.Bundle; import android.view.View; import android.widget.EditText; public class MainActivity extends AppCompatActivity { @Override protected void onCreate(Bundle savedInstanceState) { super.onCreate(savedInstanceState); setContentView(R.layout.activity_main); } //increment balls by 1 public void addBall(View view) { EditText ballText = (EditText) findViewById(R.id.balls); String ballStr = ballText.getText().toString(); EditText strikeText = (EditText) findViewById(R.id.balls); String strikeStr = strikeText.getText().toString(); int balls = Integer.parseInt(ballStr); balls = balls + 1; ballText.setText(balls); } //increment strike by 1 public void addStrike(View view) { EditText strikeText = (EditText) findViewById(R.id.strikes); String strikeStr = strikeText.getText().toString(); int strikes = Integer.parseInt(strikeStr); strikes = strikes + 1; strikeText.setText(strikes); } public void reset(View view) { EditText ballText = (EditText) findViewById(R.id.balls); EditText strikeText = (EditText) findViewById(R.id.strikes); ballText.setText(0); strikeText.setText(0); } public void exit(View view) { Intent homeIntent = new Intent(Intent.ACTION_MAIN); homeIntent.addCategory( Intent.CATEGORY_HOME ); homeIntent.setFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP); startActivity(homeIntent); } }
